def __resource_lookup(name)
  resource = root!.resources[name]
  if(resource.nil?)
    name
  else
    "#{resource.type}.#{name}"
  end
end

def _fn_format(*args)
  src = ::Kernel.__callee__.to_s
  src = ::Bogo::Utility.camel(src.sub(/(^_|\!$)/, ''), false)
  ::SparkleFormation::TerraformStruct.new(src, *args)
end
